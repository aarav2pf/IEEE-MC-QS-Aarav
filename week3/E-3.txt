class MyCircularQueue{
public:
    int* arr;
    int front,rear,size,max;
    MyCircularQueue(int k) {
        max=k;
        arr=new int[k];
        front=0;
        rear=-1;
        size=0;
    }
    bool enQueue(int value) {
        if(isFull())return false;
        rear=(rear+1)%max;
        arr[rear]=value;
        size++;
        return true;
    }
    bool deQueue() {
        if(isEmpty()) return false;
        front=(front+1)%max;
        size--;
        return true;
    }
    int Front() {
        if(isEmpty()) return -1;
        return arr[front];
    }
    int Rear() {
        if(isEmpty()) return -1;
        return arr[rear];
    }
    bool isEmpty() {
        return size==0;
    }
    bool isFull() {
        return size==max;
    }
};
