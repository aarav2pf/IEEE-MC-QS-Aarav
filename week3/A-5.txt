class Solution{
public:
    Node* connect(Node* root) {
        if(root==NULL) return root;
        Node* leftmost=root;
        while(leftmost->left!=NULL) {
            Node* head=leftmost;
            while(head!=NULL) {
                head->left->next=head->right;
                if(head->next!=NULL) {
                    head->right->next=head->next->left;
                }
                head=head->next;
            }
            leftmost=leftmost->left;
        }
        return root;
    }
};
