class Solution {
public:
    ListNode* insertionSortList (ListNode* head) {
        if (head==NULL||head->next==NULL) return head;
        ListNode* temp=new ListNode(0);
        ListNode* curr=head;
        while (curr!=NULL) {
            ListNode* nextNode=curr->next;
            ListNode* prev=temp;
            while (prev->next!=NULL&&prev->next->val<curr->val) {
                prev=prev->next;
            }
            curr->next=prev->next;
            prev->next=curr;
            curr=nextNode;
        }
        return temp->next;
    }
};
